import { Component, OnInit } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { CommonModule } from '@angular/common';

/**
 * Imágenes de cabecera
 */
export interface Tile {
  /**
   * Número de columnas
   */
  cols: number;
  /**
   * Número de filas
   */
  rows: number;
  /**
   * Fuente/ruta de la imagen
   */
  src: string;
}
